1   package ch.hsr.se2p.mrt.activities;
2   
3   import android.content.Context;
4   import android.text.InputType;
5   import android.util.AttributeSet;
6   import android.view.View;
7   import android.widget.ArrayAdapter;
8   import android.widget.AutoCompleteTextView;
9   import android.widget.ImageButton;
10  import android.widget.LinearLayout;
11  import ch.hsr.se2p.mrt.models.Customer;
12  
13  /**
14   * Creates a combobox with a autocomplete text and a button
15   */
16  public class MRTAutocompleteSpinner extends LinearLayout {
17      private AutoCompleteTextView textView;
18  
19      public MRTAutocompleteSpinner(Context context, AttributeSet attributeSet) {
20          super(context, attributeSet);
21          initControls();
22      }
23  
24      private void initControls() {
25          setOrientation(HORIZONTAL);
26          setLayoutParams(new LayoutParams(LayoutParams.FILL_PARENT, LayoutParams.WRAP_CONTENT));
27          initTextView(getContext());
28          initButton();
29      }
30  
31      private void initButton() {
32          ImageButton button = new ImageButton(getContext());
33          button.setImageResource(android.R.drawable.arrow_down_float);
34          button.setOnClickListener(getButtonListener());
35          this.addView(button, new LayoutParams(LayoutParams.WRAP_CONTENT, LayoutParams.WRAP_CONTENT));
36      }
37  
38      private OnClickListener getButtonListener() {
39          return new OnClickListener() {
40              @Override
41              public void onClick(View arg0) {
42                  textView.showDropDown();
43              }
44          };
45      }
46  
47      private void initTextView(Context context) {
48          textView = new AutoCompleteTextView(context);
49          textView.setSingleLine();
50          textView.setInputType(getInputTypeForTextView());
51          textView.setRawInputType(InputType.TYPE_TEXT_VARIATION_PASSWORD);
52          textView.setHint("Kunde");
53          addView(textView, new LayoutParams(LayoutParams.WRAP_CONTENT, LayoutParams.WRAP_CONTENT, 1));
54      }
55  
56      private int getInputTypeForTextView() {
57          return InputType.TYPE_CLASS_TEXT | InputType.TYPE_TEXT_VARIATION_NORMAL | InputType.TYPE_TEXT_FLAG_CAP_SENTENCES
58                  | InputType.TYPE_TEXT_FLAG_AUTO_COMPLETE | InputType.TYPE_TEXT_FLAG_AUTO_CORRECT;
59      }
60  
61      protected void setAdapter(ArrayAdapter<Customer> customers) {
62          textView.setAdapter(customers);
63      }
64  
65      protected String getText() {
66          return textView.getText().toString();
67      }
68  
69      protected void resetText() {
70          setText("");
71      }
72  
73      protected void setText(String text) {
74          textView.setText(text);
75      }
76  }
