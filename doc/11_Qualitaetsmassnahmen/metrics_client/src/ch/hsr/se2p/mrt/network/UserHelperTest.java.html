10      public void testLogin() {
11          User u = new User();
12          String firstName = "Peter", lastName = "Muster", email = "peter@muster.ch";
13          int id = 77;
14          Timestamp updatedAt = ISO8601DateParserHelperForTests.formatAndParseToTimestamp(new Timestamp(System.currentTimeMillis() - 1000 * 60 * 60));
15          expectedResultFromTransmitter("{\"field_worker\":{\"first_name\":\"" + firstName + "\",\"last_name\":\"" + lastName + "\",\"email\":\""
16                  + email + "\",\"id\":" + id + ",\"updated_at\":\"" + ISO8601DateParserHelperForTests.toString(updatedAt) + "\"}}");
17          UserHelper userHelper = new UserHelper(httpHelper);
18          assertTrue(userHelper.login("validlogin", "validpassword", u));
19          assertEquals(firstName, u.getFirstName());
20          assertEquals(lastName, u.getLastName());
21          assertEquals(email, u.getEmail());
22          assertEquals(id, u.getRailsId());
23          assertEquals(updatedAt, u.getUpdatedAt());
24      }
