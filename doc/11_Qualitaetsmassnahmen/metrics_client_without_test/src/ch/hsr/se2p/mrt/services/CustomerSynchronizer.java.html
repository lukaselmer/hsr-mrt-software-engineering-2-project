3   import java.sql.SQLException;

7   import android.util.Log;
8   import ch.hsr.se2p.mrt.application.MRTApplication;
9   import ch.hsr.se2p.mrt.database.DatabaseHelper;
10  import ch.hsr.se2p.mrt.interfaces.Receivable;
11  import ch.hsr.se2p.mrt.models.Customer;
12  import ch.hsr.se2p.mrt.models.GpsPosition;

16  import com.j256.ormlite.dao.Dao;

18  /**
19   * Synchronizes the clients customer data with the customer data on the server.
20   */
21  class CustomerSynchronizer implements Synchronizer {
22      private static final String TAG = CustomerSynchronizer.class.getSimpleName();
23  
24      private final DatabaseHelper databaseHelper;
25      private final MRTApplication mrtApplication;
26  
27      public CustomerSynchronizer(DatabaseHelper databaseHelper, MRTApplication mrtApplication) {
28          this.databaseHelper = databaseHelper;
29          this.mrtApplication = mrtApplication;
30      }
31  
32      @Override
33      public void synchronize() {
34          Log.d(TAG, "Synchronizing customers");
35          try {
36              Dao<Customer, Integer> dao = databaseHelper.getCustomerDao();
37              List<Receivable> receivables = new ArrayList<Receivable>(dao.queryForAll());
38              synchronizeCustomers(dao, new CustomerHelper(mrtApplication.getHttpHelper()), receivables);
39          } catch (SQLException e) {
40              Log.e(TAG, "Database error", e);
41          } catch (Exception e) {
42              Log.e(TAG, "Customer sync error", e);
43          }
44      }
45  
46      protected void synchronizeCustomers(Dao<Customer, Integer> dao, CustomerHelper ch, List<Receivable> receivables) throws SQLException {
47          try {
48              if (ch.synchronize(receivables, Customer.class)) {
49                  for (Receivable receivable : receivables) {
50                      processCustomer(dao, (Customer) receivable);
51                  }
52              }
53          } catch (SynchronizationException e) {
54              Log.e(TAG, "Error synchronizing customers", e);
55          }
56      }
57  
58      protected void processCustomer(Dao<Customer, Integer> dao, Customer c) throws SQLException {
59          try {
60              if (handleCreation(dao, c))
61                  return;
62              if (handleDeletion(dao, c))
63                  return;
64              else
65                  handleUpdate(dao, c);
66          } finally {
67              c.setChanged(false);
68          }
69      }
70  
71      private boolean existingCustomer(Customer c) {
72          return c.getId() != null && c.getId() > 0;
73      }
74  
75      private void handleUpdate(Dao<Customer, Integer> dao, Customer c) throws SQLException {
76          if (!c.hasChanged())
77              return;
78          updatePosition(c);
79          Log.d(TAG, "Updating " + c);
80          dao.update(c);
81      }
82  
83      private void updatePosition(Customer c) throws SQLException {
84          handleGpsPosition(c);
85          if (c.getGpsPosition() != null) {
86              databaseHelper.getGpsPositionDao().create(c.getGpsPosition());
87              c.setGpsPositionId(c.getGpsPosition().getId());
88          }
89      }
90  
91      private boolean handleCreation(Dao<Customer, Integer> dao, Customer c) throws SQLException {
92          if (existingCustomer(c)) {
93              return false;
94          }
95          Dao<GpsPosition, Integer> positionDao = databaseHelper.getGpsPositionDao();
96          if (c.getGpsPosition() != null) {
97              positionDao.create(c.getGpsPosition());
98              c.setGpsPositionId(c.getGpsPosition().getId());
99          }
100         Log.d(TAG, "Creating " + c);
101         dao.create(c);
102         return true;
103     }
104 
105     private boolean handleDeletion(Dao<Customer, Integer> dao, Customer c) throws SQLException {
106         if (!c.hasChanged())
107             return false;
108         handleGpsPosition(c);
109         if (!c.isDeleted() && existingCustomer(c)) {
110             Log.d(TAG, "Deleting " + c);
111             dao.delete(c);
112             return true;
113         }
114         return false;
115     }
116 
117     private void handleGpsPosition(Customer c) throws SQLException {
118         Dao<GpsPosition, Integer> positionDao = databaseHelper.getGpsPositionDao();
119         if (c.hasGpsPosition()) {
120             GpsPosition oldPosition = positionDao.queryForId(c.getGpsPositionId());
121             positionDao.delete(oldPosition);
122         }
123     }
124 }
