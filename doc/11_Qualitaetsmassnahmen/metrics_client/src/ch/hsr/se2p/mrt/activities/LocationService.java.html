1   package ch.hsr.se2p.mrt.activities;

3   import android.location.Criteria;
4   import android.location.Location;
5   import android.location.LocationListener;
6   import android.location.LocationManager;
7   import ch.hsr.se2p.mrt.models.GpsPosition;
8   
9   class LocationService {
10      private final LocationManager locationManager;
11      private final LocationListener locationListener = new LocationListenerAdapter() {
12          @Override
13          public void onLocationChanged(Location location) {
14              currentLocation = location;
15              currentGPSPosition = location == null ? null : new GpsPosition(location);
16              locationChangedAction.run();
17          }
18      };
19      private final Runnable locationChangedAction;
20      protected Location currentLocation;
21      private GpsPosition currentGPSPosition;
22  
23      protected LocationService(LocationManager locationManager, Runnable locationChangedAction) {
24          this.locationManager = locationManager;
25          this.locationChangedAction = locationChangedAction;
26          currentLocation = locationManager.getLastKnownLocation(getLocationProvider());
27          // Update location maximal every 10 seconds
28          locationManager.requestLocationUpdates(getLocationProvider(), 10 * 1000, 0, locationListener);
29      }
30  
31      protected Location getCurrentLocation() {
32          return currentLocation;
33      }
34  
35      protected GpsPosition getCurrentGPSPosition() {
36          return currentGPSPosition;
37      }
38  
39      private String getLocationProvider() {
40          Criteria criteria = new Criteria();
41          criteria.setAccuracy(Criteria.ACCURACY_FINE);
42          criteria.setAltitudeRequired(false);
43          criteria.setBearingRequired(false);
44          criteria.setCostAllowed(true);
45          criteria.setPowerRequirement(Criteria.POWER_LOW);
46          return locationManager.getBestProvider(criteria, true);
47      }
48  
49      protected void stop() {
50          locationManager.removeUpdates(locationListener);
51      }
52  
53  }
