9   import ch.hsr.se2p.mrt.interfaces.Confirmable;
10  import ch.hsr.se2p.mrt.interfaces.Transmittable;

12  public class TimeEntryHelper {
13      private static final String TAG = TimeEntryHelper.class.getSimpleName();
14  
15      private HttpHelper httpHelper;
16  
17      public TimeEntryHelper(HttpHelper httpHelper) {
18          this.httpHelper = httpHelper;
19      }
20  
21      public boolean transmit(Transmittable transmittable) {
22          if (transmittable.isTransmitted())
23              return true;
24          try {
25              JSONObject j = transmittable.toJSONObject();
26              String ret = httpHelper.doHttpPost(getJSONParameters(j), NetworkConfig.TIME_ENTRY_CREATE_URL);
27              return transmittable.processTransmission(new JSONObject(ret));
28          } catch (Exception e) {
29              // Request failed, pass
30              Log.w(TAG, "Request failed", e);
31          }
32          return false;
33      }
34  
35      private Object getJSONParameters(JSONObject jsonObject) throws JSONException {
36          return new JSONObject().put("time_entry", jsonObject);
37      }
38  
39      public boolean confirm(Confirmable confirmable) {
40          try {
41              String ret = httpHelper.doHttpPost(confirmable.toJSONObject(),
42                      String.format(NetworkConfig.TIME_ENTRY_CONFIRM_URL, confirmable.getIdOnServer()));
43              return confirmable.processConfirmation(new JSONObject(ret));
44          } catch (JSONException e) {
45              // Request failed, pass
46          } catch (NullPointerException e) {
47              // Request failed, pass
48          } catch (IOException e) {
49              // Request failed, pass
50          }
51          return false;
52      }
53  }
