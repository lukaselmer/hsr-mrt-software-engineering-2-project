3   import android.content.Intent;
4   import android.os.Bundle;
5   import android.preference.PreferenceManager;
6   import android.view.View;
7   import android.view.View.OnClickListener;
8   import android.widget.Button;
9   import android.widget.CheckBox;
10  import android.widget.EditText;
11  import ch.hsr.se2p.mrt.R;
12  import ch.hsr.se2p.mrt.application.MRTApplication;
13  import ch.hsr.se2p.mrt.database.DatabaseHelper;
14  import ch.hsr.se2p.mrt.network.UserHelper;
15  
16  import com.j256.ormlite.android.apptools.OrmLiteBaseActivity;
17  
18  /**
19   * Performs the user login. Requires email and password. Preferences saving is available. Shows warnings when data is incomplete or incorrect.
20   */
21  public class LoginActivity extends OrmLiteBaseActivity<DatabaseHelper> {
22      private EditText editEmail;
23      private EditText editPassword;
24      private CheckBox chbxSaveLogin;
25  
26      private MRTApplication mrtApplication;
27  
28      /** Called when the activity is first created. */
29      @Override
30      public void onCreate(Bundle savedInstanceState) {
31          super.onCreate(savedInstanceState);
32          setContentView(R.layout.login);
33  
34          initApplication();
35          initData();
36          checkPreferencesForAutoLogin();
37  
38          createClickListener((Button) findViewById(R.id.loginButton));
39      }
40  
41      private void initApplication() {
42          mrtApplication = (MRTApplication) getApplication();
43          mrtApplication.setPreferences(PreferenceManager.getDefaultSharedPreferences(this));
44      }
45  
46      private void initData() {
47          editEmail = (EditText) findViewById(R.id.editEmail);
48          editPassword = (EditText) findViewById(R.id.editPassword);
49          chbxSaveLogin = (CheckBox) findViewById(R.id.chbxSaveLogin);
50      }
51  
52      private void createClickListener(Button loginBtn) {
53          loginBtn.setOnClickListener(new OnClickListener() {
54              @Override
55              public void onClick(final View v) {
56                  checkLoginData();
57              }
58          });
59      }
60  
61      private void checkPreferencesForAutoLogin() {
62          if (mrtApplication.mayLogin()) {
63              processLogin(mrtApplication.getEmail(), mrtApplication.getPassword());
64          }
65      }
66  
67      private void checkLoginData() {
68          if (editEmail.getText().length() > 0 && editPassword.getText().length() > 0) {
69              processLogin(editEmail.getText().toString(), editPassword.getText().toString());
70          } else {
71              ActivityHelper.displayAlertDialog("Fehler", "Bitte Emailadresse und Passwort angeben!", LoginActivity.this);
72          }
73      }
74  
75      private void processLogin(String email, String password) {
76          if (new UserHelper(mrtApplication.getHttpHelper()).login(email, password, mrtApplication.getCurrentUser())) {
77              mrtApplication.login(email, password, chbxSaveLogin.isChecked());
78              switchToTimeEntryActivity();
79          } else {
80              ActivityHelper.displayAlertDialog(null, "Anmeldung schlug fehl!", this);
81              editPassword.setText("");
82          }
83      }
84  
85      private void switchToTimeEntryActivity() {
86          this.startActivity(new Intent(this, TimeEntryActivity.class));
87          finish();
88      }
89  }
