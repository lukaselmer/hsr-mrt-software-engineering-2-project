1   package ch.hsr.se2p.mrt.activities;
2   
3   import java.sql.Timestamp;
4   import java.util.List;
5   
6   import android.widget.Spinner;
7   import android.widget.TextView;
8   import ch.hsr.se2p.mrt.models.Customer;
9   import ch.hsr.se2p.mrt.models.TimeEntry;
10  import ch.hsr.se2p.mrt.models.TimeEntryType;
11  
12  class Measurement {
13      private TimeEntry timeEntry;
14      private boolean started = false;
15  
16      public Measurement() {
17          this(true);
18      }
19  
20      public Measurement(boolean start) {
21          if (start)
22              start();
23      }
24  
25      private void start() {
26          started = true;
27          timeEntry = new TimeEntry(new Timestamp(System.currentTimeMillis()));
28      }
29  
30      public TimeEntry getTimeEntry() {
31          return timeEntry;
32      }
33  
34      public TimeEntry stop(Spinner spinnerTimeEntryTypes, TextView textView, Integer gpsPositionId, MRTAutocompleteSpinner comboboxCustomers,
35              List<Customer> customers) {
36          started = false;
37          timeEntry.setTimeStop(new Timestamp(System.currentTimeMillis()));
38          setTimeEntryType(spinnerTimeEntryTypes);
39          timeEntry.setDescription(textView.getText().toString());
40          timeEntry.setGpsPositionId(gpsPositionId);
41          setCustomer(comboboxCustomers, customers);
42          return timeEntry;
43      }
44  
45      private void setCustomer(MRTAutocompleteSpinner comboboxCustomers, List<Customer> customers) {
46          if (comboboxCustomers.getText().length() != 0 && getCustomer(comboboxCustomers, customers) != null)
47              getTimeEntry().setCustomerId(getCustomer(comboboxCustomers, customers).getId());
48      }
49  
50      private Customer getCustomer(MRTAutocompleteSpinner comboboxCustomers, List<Customer> customers) {
51          String textCustomer = comboboxCustomers.getText();
52          if (textCustomer.length() == 0)
53              return null;
54          for (Customer customer : customers) {
55              if (customer.toString().equals(textCustomer)) {
56                  return customer;
57              }
58          }
59          return null;
60      }
61  
62      private void setTimeEntryType(Spinner spinnerTimeEntryTypes) {
63          if (spinnerTimeEntryTypes.getSelectedItemPosition() != 0)
64              timeEntry.setTimeEntryTypeId(((TimeEntryType) spinnerTimeEntryTypes.getSelectedItem()).getId());
65      }
66  
67      public boolean isStarted() {
68          return started;
69      }
70  }
