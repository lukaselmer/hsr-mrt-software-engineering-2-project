15  public class TimeEntryType implements Receivable, Comparable<TimeEntryType> {
16      @DatabaseField(generatedId = true)

18      @DatabaseField
19      private int railsId = 0;
20      @DatabaseField
21      private String name;
22      @DatabaseField
23      private long createdAt = 0;

27      private long updatedAt = 0;
28

30  
31      public TimeEntryType() {
32          // Needed for ormlite
33      }

35      public TimeEntryType(Integer id, String name) {

37          this.name = name;
38      }
39  
40      @Override
41      public boolean fromJSON(JSONObject timeEntryTypeObj) throws JSONException {
42          int railsId = timeEntryTypeObj.getInt("id");
43          if (railsId <= 0)
44              return false;
45          this.railsId = railsId;
46          parseMembersFromJson(timeEntryTypeObj);
47          if (deleted)
48              validUntil = ISO8601DateParser.parse(timeEntryTypeObj.getString("valid_until")).getTime();
49          return true;
50      }
51  
52      private void parseMembersFromJson(JSONObject timeEntryTypeObj) throws JSONException {
53          name = timeEntryTypeObj.getString("description");
54          createdAt = ISO8601DateParser.parse(timeEntryTypeObj.getString("created_at")).getTime();
55          updatedAt = ISO8601DateParser.parse(timeEntryTypeObj.getString("updated_at")).getTime();
56          deleted = !timeEntryTypeObj.isNull("valid_until");
57      }
58  
59      public Timestamp getCreatedAt() {
60          return new Timestamp(createdAt);
61      }
62  
63      public Integer getId() {
64          return id;
65      }
66  
67      @Override
68      public int getIdOnServer() {
69          return railsId;
70      }
71  
72      public String getName() {
73          return name;
74      }
75  
76      @Override
77      public Timestamp getUpdatedAt() {

79      }
80  
81      public Timestamp getValidUntil() {
82          return new Timestamp(validUntil);
83      }
84  
85      public boolean isDeleted() {
86          return deleted;
87      }
88

90      public String toString() {
91          return name;
92      }
