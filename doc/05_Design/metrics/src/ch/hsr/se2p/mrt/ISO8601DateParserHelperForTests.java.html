11  public class ISO8601DateParserHelperForTests {
12      public static Date parse(String input) {
13          // NOTE: SimpleDateFormat uses GMT[-+]hh:mm for the TZ which breaks
14          // things a bit. Before we go on we have to repair this.
15          SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssz");
16  
17          // this is zero time so we need to add that TZ indicator for
18          if (input.endsWith("Z")) {
19              input = input.substring(0, input.length() - 1) + "GMT-00:00";
20          } else {
21              int inset = 6;
22              String s0 = input.substring(0, input.length() - inset);
23              String s1 = input.substring(input.length() - inset, input.length());
24              input = s0 + "GMT" + s1;
25          }
26          try {
27              return df.parse(input);
28          } catch (ParseException e) {
29              Log.w(ISO8601DateParserHelperForTests.class.getSimpleName(), "Unparsable date", e);
30              return new Date(0);
31          }
32      }
33  
34      public static String toString(Date date) {
35          SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssz");
36          TimeZone tz = TimeZone.getTimeZone("UTC");
37          df.setTimeZone(tz);
38          String output = df.format(date);
39          int inset0 = 9;
40          int inset1 = 6;
41          String s0 = output.substring(0, output.length() - inset0);
42          String s1 = output.substring(output.length() - inset1, output.length());
43          String result = s0 + s1;
44          result = result.replaceAll("UTC", "+00:00");
45          return result;
46      }
47  
48      public static Date formatAndParse(Date date) {
49          return parse(toString(date));
50      }
51  
52      public static Timestamp formatAndParseToTimestamp(Date date) {
53          return new Timestamp(parse(toString(date)).getTime());
54      }
