14      private static final String PHONE = "+41 444 44 44", FIRST_NAME = "Test", LAST_NAME = "customer";

19      @Override
20      protected void setUp() throws Exception {
21          super.setUp();
22          customer = new Customer();
23          timeStamp = new Timestamp(System.currentTimeMillis());
24          customerObj = new JSONObject().put("customer", getCustomerJSON());
25      }
26  
27      private JSONObject getCustomerJSON() throws JSONException {
28          return new JSONObject().put("id", 1).put("first_name", FIRST_NAME).put("last_name", LAST_NAME).put("phone", PHONE)
29                  .put("updated_at", ISO8601DateParser.format(timeStamp));
30      }
31  
32      public void testInitialCustomer() {
33          assertEquals(new Integer(0), customer.getId());
34          assertEquals(0, customer.getIdOnServer());
35          assertNull(customer.getFirstName());
36          assertNull(customer.getLastName());
37          assertNull(customer.getPhone());
38          assertNull(customer.getGpsPositionId());
39          assertEquals(new Timestamp(0), customer.getUpdatedAt());
40          assertFalse(customer.isDeleted());
41      }
42  
43      public void testFromJSON() {
44          try {
45              customer.fromJSON(customerObj);
46              assertEquals(1, customer.getIdOnServer());
47              assertEquals(FIRST_NAME, customer.getFirstName());
48              assertEquals(LAST_NAME, customer.getLastName());
49              assertEquals(PHONE, customer.getPhone());
50              assertEquals(timeStamp, customer.getUpdatedAt());
51              assertEquals(FIRST_NAME + " " + LAST_NAME, customer.toString());
52          } catch (JSONException e) {
53              assert (false);
54              e.printStackTrace();
55          }
56      }

58      public void testCustomerSort() {
59          Set<Customer> set = new TreeSet<Customer>();
60          set.add(new Customer("Jasper", "Zazoo", "+41 444 44 11", null));
61          set.add(new Customer("Duelue", "Anton", "+41 444 44 22", null));
62          set.add(new Customer("The", "Zombie", "+41 444 44 33", 100.0));
63          set.add(new Customer("Klara", "Wayne", "+41 444 44 44", 499.9));
64          set.add(new Customer("Fritz", "Arnold", "+41 444 44 55", 800.0));
65          set.add(new Customer("Hans", "Peter", "+41 444 44 66", 500.0));
66          Object[] custArray = set.toArray();
67          assertEquals("Zombie", ((Customer) custArray[0]).getLastName());
68          assertEquals("Peter", ((Customer) custArray[1]).getLastName());
69          assertEquals("Wayne", ((Customer) custArray[2]).getLastName());
70          assertEquals("Arnold", ((Customer) custArray[3]).getLastName());
71          assertEquals("Anton", ((Customer) custArray[4]).getLastName());
