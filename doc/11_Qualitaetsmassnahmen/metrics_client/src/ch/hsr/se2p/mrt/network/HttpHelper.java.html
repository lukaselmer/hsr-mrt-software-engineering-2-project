3   import java.io.IOException;
4   import java.io.UnsupportedEncodingException;
5   
6   import org.apache.http.Header;
7   import org.apache.http.HttpEntity;
8   import org.apache.http.HttpResponse;
9   import org.apache.http.client.ClientProtocolException;
10  import org.apache.http.client.HttpClient;
11  import org.apache.http.client.methods.HttpEntityEnclosingRequestBase;
12  import org.apache.http.client.methods.HttpPost;
13  import org.apache.http.entity.StringEntity;
14  import org.apache.http.impl.client.DefaultHttpClient;
15  import org.apache.http.params.BasicHttpParams;
16  import org.apache.http.params.HttpConnectionParams;
17  import org.apache.http.params.HttpParams;
18  import org.apache.http.util.EntityUtils;
19  
20  import android.util.Log;
21  
22  public class HttpHelper {
23      private static final int HTTP_TIMEOUT_IN_MILLISECONDS = 10000;
24      private static final String TAG = HttpHelper.class.getSimpleName();
25      private String cookie;
26  
27      protected String doHttpPost(Object jsonObject, String url) throws IOException {
28          return doHttpRequest(url, jsonObject, getHttpPost(url));
29      }
30  
31      protected HttpEntityEnclosingRequestBase getHttpPost(String url) {
32          return new HttpPost(url);
33      }
34  
35      protected String doHttpRequest(String url, Object jsonObject, HttpEntityEnclosingRequestBase httpRequest) throws IOException {
36          HttpClient client = getHttpClient();
37          prepareRequest(url, jsonObject, httpRequest);
38          HttpResponse response = executeRequest(client, httpRequest);
39          return handleResponse(response);
40      }
41  
42      protected HttpClient getHttpClient() {
43          return new DefaultHttpClient(getHttpParams());
44      }
45  
46      protected HttpParams getHttpParams() {
47          HttpParams httpParams = new BasicHttpParams();
48          // Set the timeout in milliseconds until a connection is established.
49          HttpConnectionParams.setConnectionTimeout(httpParams, HTTP_TIMEOUT_IN_MILLISECONDS);
50          // Set the default socket timeout (SO_TIMEOUT)
51          // in milliseconds which is the timeout for waiting for data.
52          HttpConnectionParams.setSoTimeout(httpParams, HTTP_TIMEOUT_IN_MILLISECONDS);
53          return httpParams;
54      }
55  
56      protected void prepareRequest(String url, Object jsonObject, HttpEntityEnclosingRequestBase httpRequest) {
57          Log.i(TAG, "Starting HTTP Reququest: " + url);
58          try {
59              StringEntity se = new StringEntity(jsonObject.toString());
60              httpRequest.setEntity(se);
61              httpRequest.setHeader("Content-Type", "application/json");
62          } catch (UnsupportedEncodingException e) {
63              Log.e("Error", "" + e, e);
64          }
65          if (cookie != null)
66              httpRequest.addHeader("cookie", cookie);
67      }
68  
69      protected HttpResponse executeRequest(HttpClient client, HttpEntityEnclosingRequestBase httpRequest) throws IOException {
70          HttpResponse response = null;
71          Log.i(TAG, "Executing HTTP request");
72  
73          try {
74              response = client.execute(httpRequest);
75          } catch (ClientProtocolException e) {
76              Log.d("ClientProtocol", "ClientProtocolException when executing request", e);
77              return null;
78          } catch (IOException e) {
79              Log.d(TAG, "IOException when executing request", e);
80              throw e;
81          }
82          return response;
83      }
84  
85      protected String handleResponse(HttpResponse response) throws IOException {
86          Log.i(TAG, "HTTP request finished");
87          HttpEntity entity = response.getEntity();
88          Log.i(TAG, "Server answer length is: " + entity.getContentLength());
89          String responseString = EntityUtils.toString(entity);
90          Log.i(TAG, "Server answer is: " + responseString);
91  
92          Header h = response.getFirstHeader("set-cookie");
93          if (h != null)
94              cookie = h.getValue();
95          return responseString;
96      }
97  
98      protected String getCookie() {
99          return cookie;
100     }
