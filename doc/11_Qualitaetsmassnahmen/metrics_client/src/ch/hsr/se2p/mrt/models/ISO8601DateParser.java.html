1   package ch.hsr.se2p.mrt.models;
2   
3   import java.text.ParseException;
4   import java.text.SimpleDateFormat;
5   import java.util.Date;
6   import java.util.TimeZone;
7   
8   import android.util.Log;
9   
10  /**
11   * Parses and formats ISO 8601 Strings / Dates
12   */
13  class ISO8601DateParser {
14      // 2004-06-14T19:GMT20:30Z
15      // 2004-06-20T06:GMT22:01Z
16  
17      // http://www.cl.cam.ac.uk/~mgk25/iso-time.html
18      //
19      // http://www.intertwingly.net/wiki/pie/DateTime
20      //
21      // http://www.w3.org/TR/NOTE-datetime
22      //
23      // Different standards may need different levels of granularity in the date and
24      // time, so this profile defines six levels. Standards that reference this
25      // profile should specify one or more of these granularities. If a given
26      // standard allows more than one granularity, it should specify the meaning of
27      // the dates and times with reduced precision, for example, the result of
28      // comparing two dates with different precisions.
29  
30      // The formats are as follows. Exactly the components shown here must be
31      // present, with exactly this punctuation. Note that the "T" appears literally
32      // in the string, to indicate the beginning of the time element, as specified in
33      // ISO 8601.
34  
35      // Year:
36      // YYYY (eg 1997)
37      // Year and month:
38      // YYYY-MM (eg 1997-07)
39      // Complete date:
40      // YYYY-MM-DD (eg 1997-07-16)
41      // Complete date plus hours and minutes:
42      // YYYY-MM-DDThh:mmTZD (eg 1997-07-16T19:20+01:00)
43      // Complete date plus hours, minutes and seconds:
44      // YYYY-MM-DDThh:mm:ssTZD (eg 1997-07-16T19:20:30+01:00)
45      // Complete date plus hours, minutes, seconds and a decimal fraction of a
46      // second
47      // YYYY-MM-DDThh:mm:ss.sTZD (eg 1997-07-16T19:20:30.45+01:00)
48  
49      // where:
50  
51      // YYYY = four-digit year
52      // MM = two-digit month (01=January, etc.)
53      // DD = two-digit day of month (01 through 31)
54      // hh = two digits of hour (00 through 23) (am/pm NOT allowed)
55      // mm = two digits of minute (00 through 59)
56      // ss = two digits of second (00 through 59)
57      // s = one or more digits representing a decimal fraction of a second
58      // TZD = time zone designator (Z or +hh:mm or -hh:mm)
59  
60      /* Parses a ISO 8601 String to a Date */
61      protected static Date parse(String input) {
62          // NOTE: SimpleDateFormat uses GMT[-+]hh:mm for the TZ which breaks things a bit. Before we go on we have to repair this.
63          SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssz");
64          // this is zero time so we need to add that TZ indicator for
65          if (input.endsWith("Z")) {
66              input = input.substring(0, input.length() - 1) + "GMT-00:00";
67          } else {
68              int inset = 6;
69              String s0 = input.substring(0, input.length() - inset);
70              String s1 = input.substring(input.length() - inset, input.length());
71              input = s0 + "GMT" + s1;
72          }
73          return parse(input, df);
74      }
75  
76      private static Date parse(String input, SimpleDateFormat df) {
77          try {
78              return df.parse(input);
79          } catch (ParseException e) {
80              Log.w(ISO8601DateParser.class.getSimpleName(), "Unparsable date", e);
81              return new Date(0);
82          }
83      }
84  
85      /* Formats a Date to a ISO 8601 String */
86      protected static String format(Date date) {
87          SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssz");
88          df.setTimeZone(TimeZone.getTimeZone("UTC"));
89          String output = df.format(date);
90          int inset0 = 9;
91          int inset1 = 6;
92          String s0 = output.substring(0, output.length() - inset0);
93          String s1 = output.substring(output.length() - inset1, output.length());
94          String result = s0 + s1;
95          result = result.replaceAll("UTC", "+00:00");
96          return result;
97      }
98  }
