14  public class CustomerHelper {

17      public CustomerHelper(HttpHelper httpHelper) {
18          this.httpHelper = httpHelper;

21      public boolean synchronize(List<Receivable> receivables, Class<? extends Receivable> clazz) throws SynchronizationException {
22          try {
23              String ret = synchronizeRequest(receivables);
24              JSONArray arr = new JSONArray(ret);
25              if (arr.length() == 0)
26                  return false;
27              updateOrCreateReceivables(receivables, arr, clazz);
28              return true;
29          } catch (Exception e) {
30              // Request failed, throw synchronization exception
31              throw new SynchronizationException(e);
32          }
33      }
34  
35      private void updateOrCreateReceivables(List<Receivable> receivables, JSONArray arrayOfJSONReceivables, Class<? extends Receivable> clazz)
36              throws JSONException, IllegalAccessException, InstantiationException {
37          for (int i = 0; i < arrayOfJSONReceivables.length(); i++) {
38              JSONObject o = arrayOfJSONReceivables.getJSONObject(i).getJSONObject("customer");
39              Receivable r = clazz.newInstance();
40              r.fromJSON(o);
41              updateOrCreateReceivable(r, receivables, o);
42          }
43      }
44  
45      private void updateOrCreateReceivable(Receivable r, List<Receivable> receivables, JSONObject jsonObject) throws JSONException {
46          for (Receivable receivable : receivables) {
47              if (receivable.getIdOnServer() == r.getIdOnServer()) {
48                  receivable.fromJSON(jsonObject);
49                  return;
50              }
51          }
52          receivables.add(r);
53      }
54  
55      private String synchronizeRequest(List<Receivable> receivables) throws IOException, JSONException {
56          return httpHelper.doHttpPost(generateJSONRequest(receivables), NetworkConfig.SYNCHRONIZE_CUSTOMERS_URL);
57      }
58  
59      protected JSONObject generateJSONRequest(List<Receivable> receivables) throws JSONException {
60          JSONObject ret = new JSONObject();
61          Timestamp maxUpdatedAt = new Timestamp(0);
62          for (Receivable receivable : receivables) {
63              if (maxUpdatedAt.before(receivable.getUpdatedAt())) {
64                  maxUpdatedAt = receivable.getUpdatedAt();
65              }
66          }
67          ret.put("last_update", maxUpdatedAt.getTime());
68          Log.d("last_update", "Newest customer dataset is from " + maxUpdatedAt.getTime());
69  
70          return ret;
71      }
