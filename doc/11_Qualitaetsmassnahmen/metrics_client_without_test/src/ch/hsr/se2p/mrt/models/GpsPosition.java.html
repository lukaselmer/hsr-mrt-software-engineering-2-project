3   import org.json.JSONException;

6   import android.location.Location;

9   import com.j256.ormlite.field.DatabaseField;

11  /**
12   * A GpsPosition contains the geographical coordinates (latitude and longitude). Additionally, time is the time when the gpsPosition was fixed and the
13   * createdAt is the time when the gpsPosition was saved the first time.
14   */
15  public class GpsPosition {
16      private static final String TAG = TimeEntry.class.getSimpleName();
17  
18      @DatabaseField(generatedId = true)
19      private int id;
20      @DatabaseField
21      private double latitude = 0.0;
22      @DatabaseField
23      private double longitude = 0.0;
24  
25      public GpsPosition() {
26          // Needed for ormlite
27      }
28  
29      public GpsPosition(double latitude, double longitude) {
30          this(0L, latitude, longitude);
31      }
32  
33      public GpsPosition(Location location) {
34          latitude = location.getLatitude();
35          longitude = location.getLongitude();
36      }
37  
38      private GpsPosition(long time, double latitude, double longitude) {
39          this.latitude = latitude;
40          this.longitude = longitude;
41      }
42  
43      /**
44       * Calculates the distance in meters from one to another GPS Position.
45       */
46      public double distanceTo(GpsPosition otherPos) {
47          return distance(latitude, longitude, otherPos.getLatitude(), otherPos.getLongitude());
48      }
49  
50      private float distance(double lat_a, double lng_a, double lat_b, double lng_b) {
51          float[] results = { Float.MAX_VALUE };
52          Location.distanceBetween(lat_a, lng_a, lat_b, lng_b, results);
53          return results[0];
54      }
55  
56      public Integer getId() {
57          return id;
58      }
59  
60      public double getLatitude() {
61          return latitude;
62      }
63  
64      public double getLongitude() {
65          return longitude;
66      }
67  
68      public JSONObject toJSONObject() {
69          JSONObject j = new JSONObject();
70          try {
71              // j.put("time", time);
72              j.put("latitude", latitude);
73              j.put("longitude", longitude);
74          } catch (JSONException e) {
75              Log.e(TAG, "Error creating JSON Object", e);
76          }
77          return j;
78      }
79  }
