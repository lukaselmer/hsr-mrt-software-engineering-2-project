14      private static final String FIRST_NAME = "field";
15      private User user;
16      private Timestamp timeStamp;

18  
19      @Override
20      protected void setUp() throws Exception {
21          super.setUp();
22          user = new User();
23          timeStamp = new Timestamp(System.currentTimeMillis());
24          userObj = new JSONObject().put("field_worker", getUserJSON());
25      }
26  
27      private JSONObject getUserJSON() throws JSONException {
28          return new JSONObject().put("id", 1).put("first_name", FIRST_NAME).put("last_name", LAST_NAME).put("email", EMAIL)
29                  .put("updated_at", ISO8601DateParser.format(timeStamp));
30      }
31  
32      public void testInitialUser() {
33          assertNull(user.getFirstName());
34          assertNull(user.getLastName());
35          assertNull(user.getEmail());
36          assertEquals(new Timestamp(0), user.getUpdatedAt());
37      }
38  
39      public void testFromJSON() {
40          try {
41              user.fromJSON(userObj);
42              assertEquals(1, user.getIdOnServer());
43              assertEquals(FIRST_NAME, user.getFirstName());
44              assertEquals(LAST_NAME, user.getLastName());
45              assertEquals(EMAIL, user.getEmail());
46              assertEquals(ISO8601DateParser.format(timeStamp), ISO8601DateParser.format((user.getUpdatedAt())));
47          } catch (JSONException e) {
48              e.printStackTrace();
49              assert (false);
50          }
51      }
