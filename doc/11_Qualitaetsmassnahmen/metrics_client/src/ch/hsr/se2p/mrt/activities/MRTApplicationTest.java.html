1   package ch.hsr.se2p.mrt.activities;
2   
3   import android.content.SharedPreferences;
4   import android.preference.PreferenceManager;
5   import android.test.AndroidTestCase;
6   import ch.hsr.se2p.mrt.application.MRTApplication;
7   
8   public class MRTApplicationTest extends AndroidTestCase {
9   
10      private MRTApplication mrtApp;
11      private final String EMAIL = "field_worker@mrt.ch", PASSWORD = "mrt";
12  
13      @Override
14      protected void setUp() throws Exception {
15          mrtApp = new MRTApplication();
16          super.setUp();
17      }
18  
19      public void testInitialMRTApplication() {
20          assertNotNull(mrtApp.getHttpHelper());
21          assertNotNull(mrtApp.getCurrentUser());
22          assertNull(mrtApp.getEmail());
23          assertNull(mrtApp.getPassword());
24          assertNull(mrtApp.getPreferences());
25      }
26  
27      public void testLoginWithoutCredentials() {
28          mrtApp.login(EMAIL, PASSWORD, false);
29          assertEquals(EMAIL, mrtApp.getEmail());
30          assertEquals(PASSWORD, mrtApp.getPassword());
31      }
32  
33      public void testLoginWithCredentials() {
34          setPreferencesAndLogin(true);
35          SharedPreferences sp = mrtApp.getPreferences();
36          assertEquals(EMAIL, sp.getString("email", null));
37          assertEquals(PASSWORD, sp.getString("password", null));
38      }
39  
40      public void testLogout() {
41          setPreferencesAndLogin(false);
42          mrtApp.logout();
43          assertNull(mrtApp.getEmail());
44          assertNull(mrtApp.getPassword());
45      }
46  
47      private void setPreferencesAndLogin(boolean saveCredentials) {
48          mrtApp.setPreferences(PreferenceManager.getDefaultSharedPreferences(this.getContext()));
49          mrtApp.login(EMAIL, PASSWORD, saveCredentials);
50      }
