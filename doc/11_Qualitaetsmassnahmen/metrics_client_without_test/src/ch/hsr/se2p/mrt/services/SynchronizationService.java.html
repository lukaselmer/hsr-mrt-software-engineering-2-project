3   import java.util.Timer;
4   
5   import android.content.Intent;
6   import android.os.IBinder;
7   import android.util.Log;
8   import ch.hsr.se2p.mrt.database.DatabaseHelper;
9   
10  import com.j256.ormlite.android.apptools.OrmLiteBaseService;
11  
12  public class SynchronizationService extends OrmLiteBaseService<DatabaseHelper> {
13      private static final String TAG = SynchronizationService.class.getSimpleName();
14  
15      private Timer timer;
16      private SynchronizationServiceTask serviceTask;
17  
18      private static final int UPDATE_INTERVAL = 1000 * 30; // every 30 seconds
19      private static final int START_DELAY = 1000; // 1 seconds
20  
21      @Override
22      public int onStartCommand(Intent intent, int flags, int startId) {
23          Log.i("LocalService", "Received start id " + startId + ": " + intent);
24          // We want this service to continue running until it is explicitly
25          // stopped, so return sticky.
26          return START_STICKY;
27      }
28  
29      @Override
30      public IBinder onBind(Intent intent) {
31          // Not bindable, so return null
32          return null;
33      }
34  
35      @Override
36      public void onCreate() {
37          super.onCreate();
38          Log.i(TAG, "Service creating");
39          serviceTask = new SynchronizationServiceTask(this);
40          timer = new Timer("SynchronizationServiceTimer");
41          timer.schedule(serviceTask, START_DELAY, UPDATE_INTERVAL);
42      }
43  
44      @Override
45      public void onDestroy() {
46          super.onDestroy();
47          Log.i(TAG, "Service destroying");
48          timer.cancel();
49          timer = null;
50      }
51  }
