17  /**
18   * Synchronizes the time entry types of the client with the ones available on the server.
19   */
20  class TimeEntryTypeSynchronizer implements Synchronizer {
21      private static final String TAG = TimeEntryTypeSynchronizer.class.getSimpleName();
22  
23      private final DatabaseHelper databaseHelper;
24      private final MRTApplication mrtApplication;
25  
26      public TimeEntryTypeSynchronizer(DatabaseHelper databaseHelper, MRTApplication mrtApplication) {
27          this.databaseHelper = databaseHelper;
28          this.mrtApplication = mrtApplication;
29      }
30  
31      @Override
32      public void synchronize() {
33          Log.d(TAG, "Synchronizing time_entry_types");
34          try {
35              Dao<TimeEntryType, Integer> dao = databaseHelper.getTimeEntryTypeDao();
36              List<Receivable> receivables = new ArrayList<Receivable>(dao.queryForAll());
37              synchronizeTimeEntryTypes(dao, new TimeEntryTypeHelper(mrtApplication.getHttpHelper()), receivables);
38          } catch (SQLException e) {
39              Log.e(TAG, "Database error", e);
40          } catch (Exception e) {
41              Log.e(TAG, "TimeEntryType sync error", e);
42          }
43      }
44  
45      protected void synchronizeTimeEntryTypes(Dao<TimeEntryType, Integer> dao, TimeEntryTypeHelper ch, List<Receivable> receivables)
46              throws SQLException {
47          try {
48              if (ch.synchronize(receivables, TimeEntryType.class)) {
49                  for (Receivable receivable : receivables) {
50                      processTimeEntryType(dao, (TimeEntryType) receivable);
51                  }
52              }
53          } catch (SynchronizationException e) {
54              Log.e(TAG, "Error synchronizing TimeEntryTypes", e);
55          }
56      }
57  
58      protected void processTimeEntryType(Dao<TimeEntryType, Integer> dao, TimeEntryType t) throws SQLException {
59          if (handleDeletion(dao, t))
60              return;
61  
62          if (needsUpdating(t)) {
63              handleUpdate(dao, t);
64          } else {
65              handleCreation(dao, t);
66          }
67      }
68  
69      private boolean needsUpdating(TimeEntryType t) {
70          return t.getId() != null && t.getId() > 0;
71      }
72  
73      private void handleUpdate(Dao<TimeEntryType, Integer> dao, TimeEntryType t) throws SQLException {
74          Log.d(TAG, "Updating " + t);
75          dao.update(t);
76      }
77  
78      private void handleCreation(Dao<TimeEntryType, Integer> dao, TimeEntryType t) throws SQLException {
79          Log.d(TAG, "Creating " + t);
80          dao.create(t);
81      }
82  
83      private boolean handleDeletion(Dao<TimeEntryType, Integer> dao, TimeEntryType t) throws SQLException {
84          if (t.isDeleted()) {
85              Log.d(TAG, "Deleting " + t);
86              if (needsUpdating(t)) {
87                  dao.delete(t);
88                  return true;
89              }
90          }
91          return false;
92      }
93  }
