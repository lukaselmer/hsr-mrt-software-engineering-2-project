12       * 
13       * @return The newly created object
14       */
15      public JSONObject toJSONObject();
16  
17      /**
18       * @return Successfully transmitted
19       */
20      public boolean isTransmitted();
21  
22      /**
23       * Compares the hashcodes on the client to the server side. If they match, the hashcode on the server is removed. Additionally the Id on the
24       * server is set.
25       * 
26       * @param jsonObject
27       *            the object obtained by the request
28       * @return whether response was valid or not
29       * @throws JSONException
30       */
31      public boolean processTransmission(JSONObject jsonObject) throws JSONException;
32  }
