1   package ch.hsr.se2p.mrt.services;
2   
3   import java.sql.SQLException;
4   import java.util.List;
5   
6   import android.util.Log;
7   import ch.hsr.se2p.mrt.application.MRTApplication;
8   import ch.hsr.se2p.mrt.database.DatabaseHelper;
9   import ch.hsr.se2p.mrt.models.TimeEntry;
10  import ch.hsr.se2p.mrt.network.TimeEntryHelper;
11  
12  import com.j256.ormlite.dao.Dao;
13  
14  /**
15   * Transfers the time entries from the client to the server.
16   */
17  class TimeEntrySynchronizer implements Synchronizer {
18      private static final String TAG = TimeEntrySynchronizer.class.getSimpleName();
19  
20      private final DatabaseHelper databaseHelper;
21      private final MRTApplication mrtApplication;
22  
23      public TimeEntrySynchronizer(DatabaseHelper databaseHelper, MRTApplication mrtApplication) {
24          this.databaseHelper = databaseHelper;
25          this.mrtApplication = mrtApplication;
26      }
27  
28      @Override
29      public void synchronize() {
30          try {
31              Log.d(TAG, "Transmitting TimeEntries...");
32              List<TimeEntry> timeEntries = getTimeEntriesToTransmit();
33              if (timeEntries.size() == 0) {
34                  Log.d(TAG, "Transmission finished, no TimeEntries found.");
35                  return;
36              }
37              transmitTimeEnties(timeEntries);
38              Log.d(TAG, "Transmission finished");
39          } catch (SQLException e) {
40              Log.e(TAG, "Database excaeption", e);
41          }
42      }
43  
44      private void transmitTimeEnties(List<TimeEntry> timeEntries) throws SQLException {
45          TimeEntryHelper timeEntryHelper = new TimeEntryHelper(mrtApplication.getHttpHelper());
46  
47          for (TimeEntry timeEntry : timeEntries) {
48              setRelationObjectsOnTimeEntry(timeEntry);
49              Log.d(TAG, "Transmitting " + timeEntry + "...");
50              if (timeEntryHelper.transmit(timeEntry)) {
51                  handleTransmittedTimeEntry(databaseHelper.getTimeEntryDao(), timeEntryHelper, timeEntry);
52              } else
53                  break;
54          }
55      }
56  
57      private void setRelationObjectsOnTimeEntry(TimeEntry timeEntry) throws SQLException {
58          setCustomerRelationOnTimeEntry(timeEntry);
59          setGpsPositionRelationOnTimeEntry(timeEntry);
60          setTimeEntryTypeRelationOnTimeEntry(timeEntry);
61      }
62  
63      private void setTimeEntryTypeRelationOnTimeEntry(TimeEntry timeEntry) throws SQLException {
64          if (timeEntry.hasTimeEntryType())
65              timeEntry.setTimeEntryType(databaseHelper.getTimeEntryTypeDao().queryForId(timeEntry.getTimeEntryTypeId()));
66      }
67  
68      private void setGpsPositionRelationOnTimeEntry(TimeEntry timeEntry) throws SQLException {
69          if (timeEntry.hasGpsPosition())
70              timeEntry.setGpsPosition(databaseHelper.getGpsPositionDao().queryForId(timeEntry.getGpsPositionId()));
71      }
72  
73      private void setCustomerRelationOnTimeEntry(TimeEntry timeEntry) throws SQLException {
74          if (timeEntry.hasCustomer())
75              timeEntry.setCustomer(databaseHelper.getCustomerDao().queryForId(timeEntry.getCustomerId()));
76      }
77  
78      private void handleTransmittedTimeEntry(Dao<TimeEntry, Integer> timeEntryDao, TimeEntryHelper timeEntryHelper, TimeEntry timeEntry)
79              throws SQLException {
80          setTransmitted(timeEntryDao, timeEntry);
81          if (timeEntryHelper.confirm(timeEntry)) {
82              timeEntryDao.delete(timeEntry);
83              Log.d(TAG, "OK: " + timeEntry + "");
84          }
85      }
86  
87      private void setTransmitted(Dao<TimeEntry, Integer> timeEntryDao, TimeEntry timeEntry) throws SQLException {
88          if (!timeEntry.isTransmitted()) {
89              timeEntry.setTransmitted();
90              timeEntryDao.update(timeEntry);
91          }
92      }
93  
94      private List<TimeEntry> getTimeEntriesToTransmit() throws SQLException {
95          Dao<TimeEntry, Integer> timeEntryDao = databaseHelper.getTimeEntryDao();
96          return timeEntryDao.queryForAll();
97      }
98  
99  }
