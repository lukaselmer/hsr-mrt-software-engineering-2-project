103     /**
104      * Calculates the distance to the currentPosition and sets it on each customer
105      */
106     private static void calculateAndSetDistances(Dao<GpsPosition, Integer> dao, List<Customer> customers, GpsPosition currentPosition)
107             throws SQLException {
108         if (currentPosition == null)
109             return;
110         for (Customer c : customers) {
111             calculateAndSetDistances(dao, currentPosition, c);
112         }
113     }
114 
115     private static void calculateAndSetDistances(Dao<GpsPosition, Integer> dao, GpsPosition currentPosition, Customer c) throws SQLException {
116         if (!c.hasGpsPosition())
117             return;
118         GpsPosition customerPosition = dao.queryForId(c.getGpsPositionId());
119         if (customerPosition == null) {
120             c.setDistance(null);
121         } else {
122             double distance = currentPosition.distanceTo(customerPosition);
123             c.setDistance(distance <= CIRCLE_RADIUS_FOR_CUSTOMER_DROPDOWN ? distance : null);
124         }

126     }

128     private void stopTimeMeasurement() {
129         try {
130             setGPSImage(false);
131             saveTimeEntry();
132             Toast.makeText(getApplicationContext(), "Neuer Stundeneintrag wurde erstellt.", Toast.LENGTH_LONG).show();
133             updateGuiAfterMeasurement("Zeit gestoppt", "Start", Color.GREEN);
134             resetInputFields();
135             initSpinnerTimeEntryTypes();
136         } catch (SQLException e) {
137             logAndDisplaySQLException(e);
138         }
139     }

157     private TimeEntry saveTimeEntry() throws SQLException {
158         TimeEntry timeEntry = measurement.stop((Spinner) findViewById(R.id.spinnerTimeEntryType), (TextView) findViewById(R.id.txtDescription),
159                 saveGpsPosition(), ((MRTAutocompleteSpinner) findViewById(R.id.my_combo)), customers);
160         getHelper().getTimeEntryDao().create(timeEntry);
161         Log.v(TAG, "TimeEntry with ID " + timeEntry.getId() + " created");
162         return timeEntry;
163     }

165     private void sortCustomersByCurrentLocation() {
166         try {
167             calculateAndSetDistances(getHelper().getGpsPositionDao(), customers, locationService.getCurrentGPSPosition());
168             Collections.sort(customers);
169             customerAdapter.notifyDataSetChanged();
170         } catch (SQLException e) {
171             Log.e(TAG, "SQLException", e);
172         } catch (NullPointerException e) {
173             Log.e(TAG, "ORM Lite Exception", e);
174         }
175     }

177     private void startOrStopTimeMeasurement() {
178         if (measurement.isStarted()) {
179             stopTimeMeasurement();
180         } else {
181             measurement = new Measurement();
182             updateGuiAfterMeasurement("Zeit gestartet um " + new Time(System.currentTimeMillis()) + " Uhr", "Stop", Color.RED);
183         }
184     }

211             Collections.sort(timeEntryTypes);
212             timeEntryTypes.add(0, new TimeEntryType(0, "Kein Stundeneintragstyp"));

218     private Integer saveGpsPosition() throws SQLException {
219         if (locationService.getCurrentGPSPosition() == null)
220             return null;
221         Dao<GpsPosition, Integer> gpsPositionDao = getHelper().getGpsPositionDao();
222         gpsPositionDao.create(locationService.getCurrentGPSPosition());
223         return locationService.getCurrentGPSPosition().getId();
224     }
